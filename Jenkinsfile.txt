pipeline {
  agent any
  stages {
    stage('Production-deployment') {
      steps {
        input message: 'Do you approve deployment?', 
               parameters: [
                   choice(name: 'Approval', choices: ['Yes', 'No'], description: 'Approve or reject the production deployment')
               ]
        when {
          expression { params.Approval == 'Yes' } 
        }
        sh """
          echo "This is ok"
        """
      }
    }
  }
}
